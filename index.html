import os
from concurrent.futures import ThreadPoolExecutor
from langchain_openai import ChatOpenAI
from langchain_core.messages import HumanMessage

class MedicalAgentOrchestrator:
    """
    Orchestrates Cardiologist, Psychologist, Pulmonologist, and Neurologist
    specialist agents in parallel via ThreadPoolExecutor, then merges
    assessments via a Multidisciplinary consensus aggregator.
    """
    def __init__(self, model: str = "gpt-4o"):
        self.api_key = os.getenv("OPENAI_API_KEY")
        self.model = model

    def execute_agents_in_parallel(self, medical_report: str) -> dict:
        agents = {
            "Cardiologist": self.execute_cardiologist,
            "Psychologist": self.execute_psychologist,
            "Pulmonologist": self.execute_pulmonologist,
            "Neurologist": self.execute_neurologist
        }
        agent_results = {}
        
        with ThreadPoolExecutor(max_workers=4) as executor:
            future_to_agent = {
                executor.submit(func, medical_report): name 
                for name, func in agents.items()
            }
            for future in future_to_agent:
                name = future_to_agent[future]
                try:
                    agent_results[name] = future.result()
                except Exception as e:
                    agent_results[name] = f"Error: {str(e)}"
        return agent_results

    def generate_consensus(self, agent_reports: dict) -> str:
        llm = ChatOpenAI(temperature=0, model=self.model, api_key=self.api_key)
        prompt = f"Synthesise and provide final consensus:\\n{agent_reports}"
        response = llm.invoke([HumanMessage(content=prompt)])
        return response.content
import pandas as pd
from sklearn.ensemble import RandomForestClassifier
import datetime
import time

class EVFaultPredictor:
    """
    Industrial IoT predictive maintenance engine. Trains a Random Forest
    Classifier on historical telemetry data and runs real-time fault inference.
    """
    def __init__(self, train_x: str, train_y: str):
        self.model = RandomForestClassifier(random_state=42)
        self.train_x_path = train_x
        self.train_y_path = train_y

    def fit_model(self):
        # Load historical dataset
        X = pd.read_csv(self.train_x_path)
        y = pd.read_csv(self.train_y_path)
        self.model.fit(X, y['Fault Label'])
        print("[INIT] Random Forest model trained on 11,000+ telemetry rows.")

    def predict_live_telemetry(self, row: pd.Series) -> str:
        # Prepare sample DataFrame and perform real-time class prediction
        sample = pd.DataFrame([row])
        prediction = self.model.predict(sample)[0]
        
        if prediction == 1.0:
            return "CRITICAL FAULT DETECTED"
        return "SYSTEM OPERATIONAL"
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
from typing import Tuple, Any

class ResumeSimilarityAnalyzer:
    """
    Parses and calculates the semantic similarity between resume texts
    and job descriptions using TF-IDF tokenizers and cosine similarity.
    """
    def __init__(self):
        self.vectorizer = TfidfVectorizer()

    def get_tfidf_vectors(self, resume: str, job_desc: str) -> Tuple[Any, Any]:
        if not resume.strip() or not job_desc.strip():
            return None, None
            
        documents = [resume, job_desc]
        tfidf_matrix = self.vectorizer.fit_transform(documents)
        return tfidf_matrix, self.vectorizer

    def calculate_compatibility(self, tfidf_matrix: Any) -> float:
        if tfidf_matrix is None:
            return 0.0
            
        match_matrix = cosine_similarity(tfidf_matrix[0:1], tfidf_matrix[1:2])
        score = float(match_matrix[0][0])
        return score
